CREATE OR ALTER VIEW VW_FAT_ROMANEIO_PAO(
    COD_FATROMANEIOPAO,
    CEMP,
    CFIL,
    CODIGO,
    DATA_EMISSAO,
    DATA_ENTREGA,
    CODIGO_CLIENTE,
    NOME_CLIENTE,
    NUMERO_PEDIDO,
    COD_CADVENDEDOR)
AS
SELECT A.COD_FATROMANEIOPAO,
   A.CEMP,
   A.CFIL,
   A.CODIGO,
   CAST(A.DATA_EMISSAO AS timestamp),
   CAST(A.DATA_ENTREGA AS timestamp),
   B.CODigo,
   p.nome,
   F.NPED,
   CAST(A.COD_CADVENDEDOR AS VARCHAR(10))
FROM FAT_ROMANEIO_PAO A
LEFT JOIN cad_colaborador B ON B.cod_cadcolaborador = A.COD_CADCOLABORADOR
left join cad_pessoa p on p.codigo_pessoa = b.codigo_pessoa
LEFT JOIN FATPED F ON F.COD = A.COD_FATPEDIDO
;
